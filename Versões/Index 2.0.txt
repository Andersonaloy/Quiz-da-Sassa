<script>
/* =====================================================
   BANCO DE QUESTÕES
   Cada objeto tem:
   q = pergunta
   a = resposta correta
===================================================== */

/* =====================================================
   VARIÁVEIS DE CONTROLE DO JOGO
===================================================== */
let questions = []; // perguntas sorteadas
let current = 0;    // índice da pergunta atual
let score = 0;      // pontuação do jogador

/* =====================================================
   GERA PERGUNTAS ALEATÓRIAS
   - Embaralha o banco
   - Retorna X perguntas (padrão = 30)
===================================================== */
function gerarQuestoesAleatorias(qtd = 30) {
  let copia = [...bancoQuestoes]; // cria cópia do array
  copia.sort(() => Math.random() - 0.5); // embaralha
  return copia.slice(0, qtd); // pega apenas qtd perguntas
}

/* =====================================================
   INICIA O JOGO
===================================================== */
function startGame() {
  // Esconde tela inicial e mostra o quiz
  document.getElementById('startScreen').classList.add('hidden');
  document.getElementById('quizScreen').classList.remove('hidden');

  // Reinicia variáveis
  questions = gerarQuestoesAleatorias();
  current = 0;
  score = 0;

  showQuestion();
}

/* =====================================================
   MOSTRA A PERGUNTA ATUAL
===================================================== */
function showQuestion() {
  let q = questions[current];

  // Mostra a pergunta na tela
  document.getElementById('question').innerText = `${current+1}) ` + q.q;

  // Limpa respostas anteriores
  let optionsDiv = document.getElementById('options');
  optionsDiv.innerHTML = '';

  // Cria botões de resposta
  ["Leve","Média","Grave","Gravíssima"].forEach(option => {
    let btn = document.createElement('button');
    btn.innerText = option;
    btn.className = 'option-btn';

    // Quando clicar, verifica resposta
    btn.onclick = () => checkAnswer(btn, option, q.a);

    optionsDiv.appendChild(btn);
  });
}

/* =====================================================
   VERIFICA SE A RESPOSTA ESTÁ CORRETA
===================================================== */
function checkAnswer(button, selected, correct) {
  let buttons = document.querySelectorAll('.option-btn');

  // Desativa todos os botões após responder
  buttons.forEach(btn => btn.disabled = true);

  if (selected === correct) {
    button.classList.add('correct');
    document.getElementById('feedback').innerText = "✅ Correto!";
    score++;
  } else {
    button.classList.add('wrong');
    document.getElementById('feedback').innerText =
      "❌ Errado! Correta: " + correct;
  }

  // Mostra botão de próxima pergunta
  document.getElementById('nextBtn').classList.remove('hidden');
}

/* =====================================================
   VAI PARA PRÓXIMA PERGUNTA
===================================================== */
function nextQuestion() {
  current++;

  // Limpa feedback
  document.getElementById('feedback').innerText = '';
  document.getElementById('nextBtn').classList.add('hidden');

  if (current < questions.length) {
    showQuestion();
  } else {
    endGame();
  }
}

/* =====================================================
   FINALIZA O JOGO
===================================================== */
function endGame() {
  // Troca telas
  document.getElementById('quizScreen').classList.add('hidden');
  document.getElementById('endScreen').classList.remove('hidden');

  // Mostra pontuação
  document.getElementById('score').innerText =
    `Você acertou ${score} de 30 perguntas.`;

  // Mensagem baseada no desempenho
  let msg = "";
  if (score >= 25) msg = "🏆 Excelente! Nível DETRAN!";
  else if (score >= 18) msg = "👏 Muito bom!";
  else msg = "📚 Continue estudando!";

  document.getElementById('message').innerText = msg;
}
